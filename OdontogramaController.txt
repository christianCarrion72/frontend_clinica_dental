import {
  Controller,
  Get,
  Post,
  Body,
  Patch,
  Param,
  Delete,
  UseGuards,
} from '@nestjs/common';
import { ApiBearerAuth, ApiTags } from '@nestjs/swagger';
import { AuthGuard } from 'src/auth/guard/auth.guard';
import { RolesGuard } from 'src/auth/guard/roles.guard';
import { OdontogramaService } from './odontograma.service';
import { CreateOdontogramaDto } from './dto/create-odontograma.dto';
import { UpdateOdontogramaDto } from './dto/update-odontograma.dto';

@ApiTags('odontograma')
@ApiBearerAuth()
@UseGuards(AuthGuard, RolesGuard)
@Controller('odontograma')
export class OdontogramaController {
  constructor(private readonly odontogramaService: OdontogramaService) {}

  @Post()
  async create(@Body() createOdontogramaDto: CreateOdontogramaDto) {
    return await this.odontogramaService.create(createOdontogramaDto);
  }

  @Get()
  async findAll() {
    return await this.odontogramaService.findAll();
  }

  @Get(':id')
  async findOne(@Param('id') id: number) {
    return await this.odontogramaService.findOne(id);
  }

  @Patch(':id')
  async update(
    @Param('id') id: number,
    @Body() updateOdontogramaDto: UpdateOdontogramaDto,
  ) {
    return await this.odontogramaService.update(id, updateOdontogramaDto);
  }

  @Delete(':id')
  async remove(@Param('id') id: number) {
    return await this.odontogramaService.remove(id);
  }
}
