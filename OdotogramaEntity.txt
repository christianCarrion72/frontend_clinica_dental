import { OdontrogramaVersion } from 'src/odontrograma-version/entities/odontrograma-version.entity';
import { Paciente } from 'src/pacientes/entities/paciente.entity';
import {
  Column,
  CreateDateColumn,
  DeleteDateColumn,
  Entity,
  JoinColumn,
  OneToMany,
  OneToOne,
  PrimaryGeneratedColumn,
  UpdateDateColumn,
} from 'typeorm';

@Entity('odontograma')
export class Odontograma {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ type: 'jsonb' })
  json: string;

  @OneToOne(() => Paciente, (paciente) => paciente.odontograma)
  @JoinColumn({ name: 'paciente_id' })
  paciente: Paciente;

  @OneToMany(
    () => OdontrogramaVersion,
    (odontrogramaVersion) => odontrogramaVersion.odontograma,
  )
  odontrograma_versions: OdontrogramaVersion[];

  @CreateDateColumn()
  created_at: Date;

  @UpdateDateColumn()
  updated_at: Date;

  @DeleteDateColumn()
  deleted_at: Date;
}
