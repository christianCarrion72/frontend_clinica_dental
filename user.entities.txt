@Entity('roles')
export class Role {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ length: 50 })
  nombre: string;

  @Column({ length: 255 })
  descripcion: string;

  @OneToMany(() => User, (user) => user.rol)
